# 有效括号序列
```C++
class Solution 
{
public:
    bool isValid(string s) 
    {
        std::stack<char> _stack;
        for(int i = 0 ; i < s.size() ; i++)
        {
            if(s[i] == '{' || s[i] == '[' || s[i] == '(') _stack.push(s[i]);
            else
            {
                switch(s[i])
                {
                    case ')':
                        if(_stack.empty() || _stack.top() != '(') return false;
                        _stack.pop();
                        break;
                    case '}':
                        if(_stack.empty() || _stack.top() != '{') return false;
                        _stack.pop();
                        break;
                    case ']':
                        if(_stack.empty() || _stack.top() != '[') return false;
                        _stack.pop();
                        break;              
                }
            }
        }
        return _stack.empty() ?true : false;
    }
};
```
